function safeJson(value, fallback) { try { return JSON.parse(value); } catch { return fallback; } }
function adminOk(req, res) {
  const token = String(req.query?.token || req.query?.adminToken || req.get?.("x-admin-token") || "").trim();
  const expected = String(process.env.GIFT_ADMIN_TOKEN || process.env.ADMIN_TOKEN || "").trim();
  if (!expected || token === expected) return true;
  noCache(res);
  res.status(403).json({ ok: false, error: "admin_forbidden", runtimeVersion: VERSION, expectedToken: "GIFT_ADMIN_TOKEN" });
  return false;
}

function jsonBody(req, maxBytes = 1024 * 1024) {
  return new Promise((resolve, reject) => {
    const chunks = [];
    let size = 0;
    req.on("data", (chunk) => {
      size += chunk.length;
      if (size > maxBytes) {
        const error = new Error("payload_too_large");
        error.status = 413;
        reject(error);
        try { req.destroy(error); } catch {}
        return;
      }
      chunks.push(chunk);
    });
    req.on("end", () => resolve(safeJson(Buffer.concat(chunks).toString("utf8"), {})));
    req.on("error", reject);
  });
}
function mediaIsAllowedInComments(att = {}) {
  const type = String(att.type || "").toLowerCase();
  const name = String(att.name || att.fileName || att.url || att.previewUrl || "").toLowerCase();
  if (type === "image") return true;
  if (/\.(jpg|jpeg|png|webp|gif|heic|heif)(\?|#|$)/i.test(name)) return true;
  if (type === "video" || type === "file" || type === "audio") return false;
  if (/\.(pdf|doc|docx|xls|xlsx|zip|rar|mov|mp4|webm|m4v)(\?|#|$)/i.test(name)) return false;
  return true;
}
function patchCommentService(moduleExports) {
  if (!moduleExports || moduleExports.__akSp40Patched) return moduleExports;
  const oldCreate = moduleExports.createComment;
  const oldList = moduleExports.listComments;
  if (typeof oldCreate === "function") {
    moduleExports.createComment = function sp40CreateComment(args = {}) {
      const attachments = Array.isArray(args.attachments) ? args.attachments : [];
      const blocked = attachments.filter((att) => !mediaIsAllowedInComments(att));
      if (blocked.length) {
        args = {
          ...args,
          attachments: attachments.filter(mediaIsAllowedInComments),
          mediaPolicyNotice: "files_and_video_moved_to_gifts"
        };
      }
      return oldCreate.call(this, args);
    };
  }
  if (typeof oldList === "function") {
    moduleExports.listComments = function sp40ListComments(...args) {
      const list = oldList.apply(this, args);
      if (!Array.isArray(list)) return list;
      return list.map((comment) => {
        if (!Array.isArray(comment.attachments)) return comment;
        return { ...comment, attachments: comment.attachments.filter(mediaIsAllowedInComments) };
      });
    };
  }
  moduleExports.__akSp40Patched = true;
  return moduleExports;
}
