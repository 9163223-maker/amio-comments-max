'use strict';
const http=require('http');
const {spawn}=require('child_process');
const {URL}=require('url');
const fs=require('fs');
const path=require('path');

const RUNTIME='SP40.4';
const SOURCE='adminkit-SP40.4-features-1-3-prod-check';
const BASE_FILE=process.env.ADMINKIT_BASE_FILE_404||'media-core-sp4029.txt';
const PORT=Number(process.env.PORT||3000);
const INNER=Number(process.env.ADMINKIT_404_INNER_PORT||(PORT===3000?3300:PORT+3));
const ADMIN=process.env.DEBUG_ADMIN_TOKEN||process.env.GIFT_ADMIN_TOKEN||process.env.ADMIN_TOKEN||'admin';
const startedAt=new Date().toISOString();
const stats={proxied:0,proxyErrors:0,childRestarts:0,policyRequests:0,ctaBlocked:0,lastError:null,exports:0,lastExport:null};

function env(k,d=''){return String(process.env[k]??d)}
function cleanToken(v){let s=String(v||'').trim();s=s.replace(/^[ '\"]+|[ '\"]+$/g,'').trim();s=s.replace(/^Bearer\s+/i,'').trim();s=s.replace(/[\r\n\t ]+/g,'');return s}
function mask(s){s=String(s||'');return s?{present:true,length:s.length,prefix:s.slice(0,8),suffix:s.slice(-4)}:{present:false,length:0}}
function headers(){return {'cache-control':'no-store, no-cache, must-revalidate, proxy-revalidate','pragma':'no-cache','expires':'0','x-adminkit-runtime':RUNTIME}}
function text(res,code,s,ct='text/plain; charset=utf-8'){res.writeHead(code,{'content-type':ct,...headers()});res.end(s)}
function json(res,code,o){const b=Buffer.from(JSON.stringify(o));res.writeHead(code,{'content-type':'application/json; charset=utf-8','content-length':String(b.length),...headers()});res.end(b)}
function isAdmin(req){const u=new URL(req.url,'http://x');return String(u.searchParams.get('token')||'')===String(ADMIN)}
function guard(req,res,fn){if(!isAdmin(req))return json(res,403,{ok:false,error:'forbidden',runtimeVersion:RUNTIME});return fn()}
function ghCfg(){const raw=env('GITHUB_DEBUG_TOKEN',''),tok=cleanToken(raw);return{repo:env('GITHUB_DEBUG_REPO',''),branch:env('GITHUB_DEBUG_BRANCH','main'),path:env('GITHUB_DEBUG_PATH','debug/latest.json'),litePath:env('GITHUB_DEBUG_LITE_PATH','debug/latest-lite.json'),rawToken:mask(raw),normalizedToken:mask(tok),tokenChanged:raw!==tok,token:tok}}

const FEATURE_PLAN=[
 {id:1,key:'channel_connection',title:'Подключение канала',status:'prod-check',functions:['подключение канала через бота','сохранение channelId, названия канала и прав','проверка прав администратора','автоподстановка единственного канала','раздел Ваши каналы','защита от дублей каналов','Назад и Главное меню','восстановление меню после очистки чата']},
 {id:2,key:'post_comments',title:'Комментарии под постами',status:'prod-check',functions:['кнопка Комментарии под постом','открытие обсуждения нужного поста','Telegram-style интерфейс','свои комментарии справа','чужие комментарии слева','имя и аватар только у чужих','счётчик комментариев','быстрый ввод снизу','обновление без ручной перезагрузки','старая CTA отключена']},
 {id:3,key:'photo_comments_paid',title:'Фото в комментариях',status:'paid-prod-check',functions:['только фото','только для платного тарифа','free-тариф: текстовые комментарии и upsell','без видео','без файлов','без кастомного меню вложений','превью фото','сохранение фото после перезахода']}
];

function flags(){return{sectionsActive:'1,2,3',oldCtaCode:'disabled',customAttachmentMenu:'disabled',commentVideo:'disabled',commentFiles:'disabled',commentPhoto:env('FEATURE_COMMENT_PHOTO','1')==='0'?'disabled':'enabled',commentPhotoPaidOnly:'enabled',tariffMode:env('ADMINKIT_TARIFF_MODE','free')}}
function qaObj(){const g=ghCfg(),f=flags();const blockers=[];if(f.customAttachmentMenu!=='disabled')blockers.push('custom_attachment_menu_must_be_disabled');if(f.commentVideo!=='disabled')blockers.push('comment_video_must_be_disabled');if(f.commentFiles!=='disabled')blockers.push('comment_files_must_be_disabled');if(f.oldCtaCode!=='disabled')blockers.push('old_cta_must_be_disabled');return{ok:true,runtime:RUNTIME,runtimeVersion:RUNTIME,sourceMarker:SOURCE,generatedAt:Date.now(),generatedAtIso:new Date().toISOString(),startedAt,baseFile:BASE_FILE,innerPort:INNER,verdict:blockers.length?'BLOCKED':'PROD_CHECK_READY',safe:blockers.length===0,featurePlan:FEATURE_PLAN,featureFlags:f,productionRules:{commentsAttachments:'photo_only',videoInComments:false,filesInComments:false,customAttachmentMenu:false,photoCommentsPaidOnly:true,oldCtaEndpointBlocked:true},stats,githubExport:{repo:g.repo,branch:g.branch,path:g.path,litePath:g.litePath,rawToken:g.rawToken,normalizedToken:g.normalizedToken,tokenChanged:g.tokenChanged,exports:stats.exports,lastExport:stats.lastExport},blockers,sourceOfTruth:'/debug/qa-lite?token=admin'}}
function qaLite(){const q=qaObj(),f=q.featureFlags;return[`OK: ${q.verdict}`,`runtime: ${RUNTIME}`,`sourceMarker: ${SOURCE}`,`base: ${BASE_FILE}`,`sectionsActive: ${f.sectionsActive}`,`1 channelConnection: prod-check`,`2 postComments: prod-check`,`3 photoCommentsPaid: prod-check`,`oldCtaCode: ${f.oldCtaCode}`,`customAttachmentMenu: ${f.customAttachmentMenu}`,`commentPhoto: ${f.commentPhoto}`,`commentPhotoPaidOnly: ${f.commentPhotoPaidOnly}`,`commentVideo: ${f.commentVideo}`,`commentFiles: ${f.commentFiles}`,`tariffMode: ${f.tariffMode}`,`proxied: ${stats.proxied}`,`policyRequests: ${stats.policyRequests}`,`ctaBlocked: ${stats.ctaBlocked}`,`proxyErrors: ${stats.proxyErrors}`,`childRestarts: ${stats.childRestarts}`,`lastError: ${stats.lastError||'none'}`,`exports: ${stats.exports}`,`lastExport: ${stats.lastExport?JSON.stringify(stats.lastExport):'none'}`,`sourceOfTruth: /debug/qa-lite?token=admin`].join('\n')+'\n'}

async function putFile(repo,branch,filePath,content,message,token){const api='https://api.github.com/repos/'+repo+'/contents/'+encodeURIComponent(filePath).replace(/%2F/g,'/');const hdr={authorization:'Bearer '+token,'content-type':'application/json','user-agent':'adminkit-debug-export'};let sha=null;try{const r=await fetch(api+'?ref='+encodeURIComponent(branch),{headers:hdr});if(r.ok){const j=await r.json();sha=j.sha||null}}catch(_){}const body={message,branch,content:Buffer.from(content).toString('base64')};if(sha)body.sha=sha;const r=await fetch(api,{method:'PUT',headers:hdr,body:JSON.stringify(body)});const j=await r.json().catch(()=>({}));if(!r.ok)throw new Error('github_put_'+r.status+':'+JSON.stringify(j).slice(0,220));return j}
async function doExport(){const g=ghCfg();if(!g.token||!g.repo)throw new Error('github_debug_env_missing');await putFile(g.repo,g.branch,g.path,JSON.stringify(qaObj(),null,2),'Update Adminkit SP40.4 QA debug',g.token);await putFile(g.repo,g.branch,g.litePath,qaLite(),'Update Adminkit SP40.4 QA lite debug',g.token);stats.exports++;stats.lastExport={atIso:new Date().toISOString(),repo:g.repo,branch:g.branch,path:g.path,litePath:g.litePath};return stats.lastExport}

function proxy(req,res){return new Promise(resolve=>{const inner=http.request({hostname:'127.0.0.1',port:INNER,method:req.method,path:req.url,headers:{...req.headers,host:'127.0.0.1:'+INNER},timeout:35000},r=>{const a=[];r.on('data',c=>a.push(c));r.on('end',()=>{const b=Buffer.concat(a);const h={...r.headers,'x-adminkit-runtime':RUNTIME};delete h['transfer-encoding'];delete h.connection;h['content-length']=String(b.length);res.writeHead(r.statusCode||200,h);res.end(b);resolve()})});inner.on('timeout',()=>inner.destroy(new Error('proxy_timeout')));inner.on('error',e=>{stats.proxyErrors++;stats.lastError=e.message;json(res,502,{ok:false,runtimeVersion:RUNTIME,error:'proxy_failed',message:e.message});resolve()});req.pipe(inner)})}

let child=null,stopping=false;function startChild(){const filePath=path.join(process.cwd(),BASE_FILE);if(!fs.existsSync(filePath)){stats.lastError='base_file_missing:'+BASE_FILE;return}child=spawn(process.execPath,[filePath],{env:{...process.env,PORT:String(INNER),ADMINKIT_PARENT_RUNTIME:RUNTIME,FEATURE_ATTACHMENT_MENU:'0',ATTACHMENT_MENU_EXPERIMENT:'0',FEATURE_COMMENT_VIDEO:'0',FEATURE_COMMENT_FILES:'0'},stdio:['ignore','inherit','inherit']});child.on('exit',(code,signal)=>{child=null;if(!stopping){stats.childRestarts++;stats.lastError='child_exit:'+code+':'+signal;setTimeout(startChild,1200)}})}
process.on('SIGTERM',()=>{stopping=true;if(child)child.kill('SIGTERM');process.exit(0)});process.on('SIGINT',()=>{stopping=true;if(child)child.kill('SIGINT');process.exit(0)});startChild();

http.createServer(async(req,res)=>{try{const u=new URL(req.url,'http://x'),p=u.pathname;if(p==='/api/promo/cta'||p==='/api/cta'||p==='/api/floating-cta'){stats.ctaBlocked++;return json(res,200,{ok:true,enabled:false,disabledBy:RUNTIME,reason:'cta_legacy_removed'})}if(p==='/api/comments/attachment-policy'||p==='/api/attachment-policy'){stats.policyRequests++;return json(res,200,{ok:true,runtimeVersion:RUNTIME,policy:{free:'text_only_with_photo_upsell',paid:'photo_only',video:false,files:false,customMenu:false,photoPaidOnly:true}})}if(p==='/public/build-marker.txt'||p==='/debug/runtime-marker-sp404')return text(res,200,`${SOURCE}\n${RUNTIME}\nbase=${BASE_FILE}\ninner=${INNER}\n`);if(p==='/debug/feature-plan'||p==='/debug/features-1-3')return guard(req,res,()=>json(res,200,{ok:true,runtimeVersion:RUNTIME,sourceMarker:SOURCE,featurePlan:FEATURE_PLAN,featureFlags:flags()}));if(p==='/debug/qa-lite'||p==='/debug/sp404')return guard(req,res,()=>text(res,200,qaLite()));if(p==='/debug/qa'||p==='/debug/qa-sp404'||p==='/debug/qa-proxy')return guard(req,res,()=>json(res,200,qaObj()));if(p==='/debug/github-export-check')return guard(req,res,()=>json(res,200,{ok:true,runtimeVersion:RUNTIME,githubExport:qaObj().githubExport}));if(p==='/debug/export-sp404'||p==='/debug/export')return guard(req,res,async()=>{try{return json(res,200,{ok:true,runtimeVersion:RUNTIME,sourceMarker:SOURCE,export:await doExport()})}catch(e){stats.lastError='export:'+e.message;return json(res,500,{ok:false,runtimeVersion:RUNTIME,error:'github_export_failed',message:e.message,githubExport:qaObj().githubExport})}});stats.proxied++;return proxy(req,res)}catch(e){stats.proxyErrors++;stats.lastError=e.message;return json(res,500,{ok:false,runtimeVersion:RUNTIME,error:e.message})}}).listen(PORT,()=>console.log(`${SOURCE} listening on ${PORT}, inner ${INNER}, base ${BASE_FILE}`));
