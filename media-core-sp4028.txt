'use strict';

// AdminKIT SP40.2.8 — duplicate post guard + debug trace.
// Safe wrapper over SP40 base. No new UX and no AI enforce.

const crypto = require('crypto');
const RUNTIME_VERSION = 'SP40.2.8';
const SOURCE_MARKER = 'adminkit-SP40.2.8-duplicate-post-guard';
const BASE_ENTRY = process.env.ADMINKIT_BASE_ENTRY || './media-core-sp4021.txt';
const ADMIN_TOKEN = process.env.GIFT_ADMIN_TOKEN || process.env.ADMIN_TOKEN || 'admin';
const TTL_MS = Number(process.env.COMMENT_DEDUPE_TTL_MS || 15000);
const WAIT_MS = Number(process.env.COMMENT_DEDUPE_PENDING_WAIT_MS || 4500);

const state = {
  startedAt: new Date().toISOString(),
  received: 0,
  created: 0,
  deduped: 0,
  ignored: 0,
  errors: 0,
  cache: new Map(),
  events: []
};

function ts(){ return Date.now(); }
function hash(v){ return crypto.createHash('sha256').update(String(v || '')).digest('hex').slice(0, 12); }
function norm(v){ return String(v || '').replace(/\s+/g, ' ').trim(); }
function q(req){ return (req && req.query && typeof req.query === 'object') ? req.query : {}; }
function b(req){ return (req && req.body && typeof req.body === 'object') ? req.body : {}; }
function h(req, name){ return req && req.headers ? (req.headers[name] || req.headers[name.toLowerCase()] || '') : ''; }
function byPath(obj, paths){
  for (const p of paths) {
    let cur = obj;
    for (const part of p.split('.')) cur = cur && typeof cur === 'object' ? cur[part] : undefined;
    if (cur !== undefined && cur !== null && cur !== '') return cur;
  }
  return '';
}
function log(type, data){
  const row = Object.assign({ id: String(ts()) + '_' + Math.random().toString(36).slice(2, 7), type, at: ts(), atIso: new Date().toISOString() }, data || {});
  state.events.push(row);
  if (state.events.length > 250) state.events.splice(0, state.events.length - 250);
  return row;
}
function purge(){
  const n = ts();
  for (const [k, v] of state.cache.entries()) if (!v || n - v.at > TTL_MS) state.cache.delete(k);
}
function payload(req){
  const body = b(req), query = q(req);
  const text = norm(byPath(body, ['text','comment','message','value','payload.text','data.text']) || byPath(query, ['text','comment','message']));
  const commentKey = String(byPath(body, ['commentKey','threadKey','postKey','key','payload.commentKey','data.commentKey']) || byPath(query, ['commentKey','threadKey','postKey','key']) || '');
  const userId = String(byPath(body, ['userId','uid','authorId','from.id','user.id','payload.userId','data.userId']) || byPath(query, ['userId','uid','authorId']) || h(req, 'x-user-id') || '');
  const clientMutationId = String(byPath(body, ['clientMutationId','mutationId','requestId','dedupeKey','payload.clientMutationId','data.clientMutationId']) || byPath(query, ['clientMutationId','mutationId','requestId','dedupeKey']) || h(req, 'x-client-mutation-id') || '');
  const attachmentHash = hash(JSON.stringify(body.attachments || body.files || body.media || []));
  return { text, commentKey, userId, clientMutationId, attachmentHash, textHash: hash(text) };
}
function dedupeKey(p){
  if (!p) return '';
  if (p.clientMutationId) return 'cmid:' + hash(p.commentKey + '|' + p.userId + '|' + p.clientMutationId);
  if (!p.text && !p.attachmentHash) return '';
  return 'auto:' + hash([p.commentKey, p.userId, p.textHash, p.attachmentHash].join('|'));
}
function admin(req){ return String(q(req).token || '') === ADMIN_TOKEN; }
function json(res, obj, code){
  if (res.headersSent) return;
  if (code) res.status(code);
  res.setHeader('cache-control', 'no-store, no-cache, must-revalidate, proxy-revalidate');
  res.setHeader('pragma', 'no-cache');
  res.setHeader('expires', '0');
  res.json(obj);
}
function wait(entry){
  if (!entry || entry.status === 'done') return Promise.resolve(entry);
  return new Promise(resolve => {
    const start = ts();
    const timer = setInterval(() => {
      if (entry.status === 'done' || ts() - start >= WAIT_MS) { clearInterval(timer); resolve(entry); }
    }, 50);
  });
}
function capture(res, entry){
  if (!res || res.__adminkitDedupeCapture) return;
  res.__adminkitDedupeCapture = true;
  const oldJson = res.json, oldSend = res.send;
  res.json = function patchedJson(body){
    try { entry.responseKind = 'json'; entry.responseBody = body; entry.statusCode = res.statusCode || 200; } catch(e) {}
    return oldJson.apply(this, arguments);
  };
  res.send = function patchedSend(body){
    try { entry.responseKind = entry.responseKind || 'send'; if (entry.responseBody === undefined) entry.responseBody = body; entry.statusCode = res.statusCode || 200; } catch(e) {}
    return oldSend.apply(this, arguments);
  };
  res.once('finish', () => {
    entry.status = 'done';
    entry.elapsedMs = ts() - entry.at;
    state.created += 1;
    log('comment_post_created', { key: entry.publicKey, commentKey: entry.payload.commentKey, userId: entry.payload.userId, textHash: entry.payload.textHash, statusCode: entry.statusCode || res.statusCode || 200, elapsedMs: entry.elapsedMs });
  });
}
function guard(req, res, next){
  try {
    purge();
    state.received += 1;
    const p = payload(req);
    const key = dedupeKey(p);
    if (!key) { state.ignored += 1; log('comment_post_no_dedupe_key', { path: req.path || req.url, payload: p }); return next(); }
    const publicKey = key.slice(0, 18);
    log('comment_post_received', { key: publicKey, path: req.path || req.url, commentKey: p.commentKey, userId: p.userId, textHash: p.textHash, textPreview: p.text.slice(0, 120), hasClientMutationId: Boolean(p.clientMutationId) });
    const existing = state.cache.get(key);
    if (existing && ts() - existing.at <= TTL_MS) {
      state.deduped += 1;
      log('comment_post_deduped', { key: publicKey, commentKey: p.commentKey, userId: p.userId, textHash: p.textHash, originalStatus: existing.status, ageMs: ts() - existing.at });
      return wait(existing).then(done => {
        const cached = done && done.responseBody;
        if (cached && typeof cached === 'object') return json(res, Object.assign({}, cached, { ok: cached.ok !== false, deduped: true, duplicateGuard: RUNTIME_VERSION }), 200);
        return json(res, { ok: true, deduped: true, duplicateGuard: RUNTIME_VERSION, pending: done && done.status !== 'done' }, 200);
      }).catch(next);
    }
    const entry = { at: ts(), status: 'pending', payload: p, publicKey };
    state.cache.set(key, entry);
    capture(res, entry);
    return next();
  } catch (e) {
    state.errors += 1;
    log('comment_post_guard_error', { message: e && e.message ? e.message : String(e) });
    return next();
  }
}
function isCommentPath(path){
  if (typeof path !== 'string') return false;
  return path === '/api/comments' || path === '/comments' || path.includes('/api/comments') || path.endsWith('/comments');
}
function patchExpress(express){
  if (!express || express.__adminkitSp4028Patched) return express;
  express.__adminkitSp4028Patched = true;
  const appProto = express.application;
  if (appProto && appProto.post && !appProto.__adminkitSp4028PostPatched) {
    appProto.__adminkitSp4028PostPatched = true;
    const oldPost = appProto.post;
    appProto.post = function patchedPost(path){
      const args = Array.prototype.slice.call(arguments);
      if (isCommentPath(path)) args.splice(1, 0, guard);
      return oldPost.apply(this, args);
    };
  }
  if (appProto && appProto.listen && !appProto.__adminkitSp4028ListenPatched) {
    appProto.__adminkitSp4028ListenPatched = true;
    const oldListen = appProto.listen;
    appProto.listen = function patchedListen(){
      try { installDebug(this); } catch(e) { log('debug_route_install_error', { message: e.message }); }
      return oldListen.apply(this, arguments);
    };
  }
  const oldRouter = express.Router;
  if (typeof oldRouter === 'function' && !oldRouter.__adminkitSp4028RouterPatched) {
    function Router(){
      const router = oldRouter.apply(this, arguments);
      if (router && router.post && !router.__adminkitSp4028PostPatched) {
        router.__adminkitSp4028PostPatched = true;
        const oldPost = router.post;
        router.post = function patchedRouterPost(path){
          const args = Array.prototype.slice.call(arguments);
          if (isCommentPath(path)) args.splice(1, 0, guard);
          return oldPost.apply(this, args);
        };
      }
      return router;
    }
    Object.assign(Router, oldRouter);
    Router.__adminkitSp4028RouterPatched = true;
    express.Router = Router;
  }
  return express;
}
function installDebug(app){
  if (!app || app.__adminkitSp4028DebugRoutes) return;
  app.__adminkitSp4028DebugRoutes = true;
  app.get('/debug/duplicate-posts', (req, res) => {
    if (!admin(req)) return json(res, { ok:false, error:'forbidden' }, 403);
    purge();
    json(res, { ok:true, runtimeVersion:RUNTIME_VERSION, sourceMarker:SOURCE_MARKER, startedAt:state.startedAt, baseEntry:BASE_ENTRY, ttlMs:TTL_MS, pendingWaitMs:WAIT_MS, stats:{ received:state.received, created:state.created, deduped:state.deduped, ignored:state.ignored, errors:state.errors, cacheSize:state.cache.size }, cache:Array.from(state.cache.values()).slice(-25).map(e => ({ key:e.publicKey, status:e.status, ageMs:ts()-e.at, commentKey:e.payload.commentKey, userId:e.payload.userId, textHash:e.payload.textHash, textPreview:e.payload.text.slice(0,80), elapsedMs:e.elapsedMs||0, statusCode:e.statusCode||0 })), events:state.events.slice(-120) });
  });
  app.get('/debug/duplicate-posts-clear', (req, res) => {
    if (!admin(req)) return json(res, { ok:false, error:'forbidden' }, 403);
    const removed = state.events.length;
    state.events = []; state.cache.clear(); state.received = 0; state.created = 0; state.deduped = 0; state.ignored = 0; state.errors = 0;
    json(res, { ok:true, runtimeVersion:RUNTIME_VERSION, sourceMarker:SOURCE_MARKER, removed });
  });
  app.get('/debug/runtime-marker-sp4028', (req, res) => {
    if (!admin(req)) return json(res, { ok:false, error:'forbidden' }, 403);
    json(res, { ok:true, serverRuntime:RUNTIME_VERSION, serverSourceMarker:SOURCE_MARKER, baseEntry:BASE_ENTRY, baseLayer:'SP40', safetyLayer:'SP40.2.8', frontendMutation:false, newUxEnabled:false, duplicateGuard:{ active:true, ttlMs:TTL_MS, pendingWaitMs:WAIT_MS } });
  });
}

try {
  const express = require('express');
  patchExpress(express);
  log('sp4028_bootstrap_started', { baseEntry: BASE_ENTRY });
  require(BASE_ENTRY);
} catch (err) {
  log('sp4028_bootstrap_failed', { message: err && err.stack ? err.stack : String(err) });
  console.error('[SP40.2.8] bootstrap failed', err);
  throw err;
}
